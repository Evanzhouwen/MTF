235  void update() override{
236  ++frame_id;
237  am.setFirstIter();
238  for(int iter_id = 0; iter_id < params.max_iters; iter_id++){
239  // extract pixel values from the current image at the latest known position of the object
240  am.updatePixVals(ssm.getPts());
241 
242  if(params.enable_spi){
243  rel_pix_diff = (am.getInitPixVals() - am.getCurrPixVals()) / max_pix_diff;
244  pix_mask = rel_pix_diff.cwiseAbs().array() < params.spi_thresh;
245  }
246  // compute pixel gradient of the current image warped with the current warp
247  ssm.updateGradPts(am.getGradOffset());
248  am.updatePixGrad(ssm.getGradPts());
249  // multiply the pixel gradient with the SSM Jacobian to get the Jacobian of pixel values w.r.t. SSM parameters
250  ssm.cmptInitPixJacobian(curr_pix_jacobian, am.getCurrPixGrad());
251 
252  if(params.jac_type == JacType::Original || params.hess_type == HessType::Original){
253  mean_pix_jacobian = (init_pix_jacobian + curr_pix_jacobian) / 2.0;
254  }
255  if(params.sec_ord_hess && params.hess_type != HessType::InitialSelf){
256  ssm.updateHessPts(am.getHessOffset());
257  am.updatePixHess(ssm.getPts(), ssm.getHessPts());
258  ssm.cmptInitPixHessian(curr_pix_hessian, am.getCurrPixHess(), am.getCurrPixGrad());
259  }
260  // compute the prerequisites for the gradient functions
261  am.updateSimilarity();
262  // update the gradient of the error norm w.r.t. current pixel values
263  am.updateCurrGrad();
264  // update the gradient of the error norm w.r.t. initial pixel values
265  am.updateInitGrad();
266 
267  switch(params.jac_type){
268  case JacType::Original:
269  // take the mean at the level of the jacobian of the pixel values wrt SSM parameters,
270  //then use this mean jacobian to compute the Jacobian of the error norm wrt SSM parameters
271  am.cmptCurrJacobian(jacobian, mean_pix_jacobian);
272  break;
273  case JacType::DiffOfJacs:
274  // compute the mean difference between the Jacobians of the error norm w.r.t. initial AND current values of SSM parameters
275  am.cmptDifferenceOfJacobians(jacobian, init_pix_jacobian, curr_pix_jacobian);
276  jacobian *= 0.5;
277  break;
278  }
279  switch(params.hess_type){
280  case HessType::InitialSelf:
281  break;
282  case HessType::Original:
283  if(params.sec_ord_hess){
284  mean_pix_hessian = (init_pix_hessian + curr_pix_hessian) / 2.0;
285  am.cmptCurrHessian(hessian, mean_pix_jacobian, mean_pix_hessian);
286  } else{
287  am.cmptCurrHessian(hessian, mean_pix_jacobian);
288  }
289  break;
290  case HessType::SumOfStd:
291  if(params.sec_ord_hess){
292  am.cmptSumOfHessians(hessian, init_pix_jacobian, curr_pix_jacobian,
293  init_pix_hessian, curr_pix_hessian);
294  } else{
295  am.cmptSumOfHessians(hessian, init_pix_jacobian, curr_pix_jacobian);
296  }
297  hessian *= 0.5;
298  break;
299  case HessType::SumOfSelf:
300  if(params.sec_ord_hess){
301  am.cmptSelfHessian(hessian, curr_pix_jacobian, curr_pix_hessian);
302  } else{
303  am.cmptSelfHessian(hessian, curr_pix_jacobian);
304  }
305  hessian = (hessian + init_self_hessian) * 0.5;
306  break;
307  case HessType::CurrentSelf:
308  if(params.sec_ord_hess){
309  am.cmptSelfHessian(hessian, curr_pix_jacobian, curr_pix_hessian);
310  } else{
311  am.cmptSelfHessian(hessian, curr_pix_jacobian);
312  }
313  break;
314  case HessType::Std:
315  if(params.sec_ord_hess){
316  am.cmptCurrHessian(hessian, curr_pix_jacobian, curr_pix_hessian);
317  } else{
318  am.cmptCurrHessian(hessian, curr_pix_jacobian);
319  }
320  break;
321  }
322  ssm_update = -hessian.colPivHouseholderQr().solve(jacobian.transpose());
323  prev_corners = ssm.getCorners();
324  ssm.compositionalUpdate(ssm_update);
325  double update_norm = (prev_corners - ssm.getCorners()).squaredNorm();
326  if(update_norm < params.epsilon){
327  break;
328  }
329  am.clearFirstIter();
330  }
331  ssm.getCorners(cv_corners_mat);
332  }
